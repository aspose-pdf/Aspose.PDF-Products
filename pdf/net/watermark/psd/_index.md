{{% blocks/products/pf/agp/code-block title="Add Watermark to PSD - C#" offSpacer="" %}}

```cs

// load existing PSD document
var document = new Aspose.Pdf.Document("template.psd");

// create an instance of TextStamp with watermark text
var textStamp = new Aspose.Pdf.TextStamp("CONFIDENTIAL")
{
    Background = true,
    XIndent = 100,
    YIndent = 100,
};
// set text properties
textStamp.TextState.Font = Aspose.Pdf.Text.FontRepository.FindFont("Arial");
textStamp.TextState.FontSize = 14.0F;
textStamp.TextState.ForegroundColor = Aspose.Pdf.Color.FromRgb(System.Drawing.Color.Aqua);

// add stamp to particular page
document.Pages[1].AddStamp(textStamp);
// save the document back on disc
document.Save("output.psd");

```
